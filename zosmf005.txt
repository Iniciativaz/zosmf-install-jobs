//ZOSMF005 JOB (MVSS),'IZUCASEC',MSGCLASS=H,CLASS=A,NOTIFY=&SYSUID
//*******************************************************************/
//* This sample job intends to help with the security setup required*/
//* for the Network Configuration Assistant plugin.  It consists of */
//* three parts:                                                    */
//*******************************************************************/
//STEP1  EXEC PGM=IKJEFT01,DYNAMNBR=99
//SYSPRINT DD SYSOUT=*
//SYSTSPRT DD SYSOUT=*
//SYSTSIN  DD *

 RDEFINE ZMFAPLA +
  IZUDFLT.ZOSMF.CONFIGURATION_ASSISTANT.CONFIGURATION_ASSISTANT +
  UACC(NONE)

 PERMIT +
   IZUDFLT.ZOSMF.CONFIGURATION_ASSISTANT.CONFIGURATION_ASSISTANT +
   CLASS(ZMFAPLA) ID(IZUUSER) ACCESS(READ)

 PERMIT +
   IZUDFLT.ZOSMF.CONFIGURATION_ASSISTANT.CONFIGURATION_ASSISTANT +
   CLASS(ZMFAPLA) ID(IZUADMIN) ACCESS(READ)

 SETROPTS RACLIST(ZMFAPLA) REFRESH

 RDEFINE OPERCMDS MVS.MCSOPER.* UACC(NONE)

 PERMIT MVS.MCSOPER.* CLASS(OPERCMDS) ID(IZUSVR) ACCESS(READ)

 RDEFINE OPERCMDS (MVS.VARY.TCPIP.OBEYFILE) UACC(NONE)

 PERMIT MVS.VARY.TCPIP.OBEYFILE CLASS(OPERCMDS) ID(IZUSVR) +
   ACCESS(CONTROL)

 RDEFINE OPERCMDS (MVS.DISPLAY.*) UACC(NONE)

 PERMIT MVS.DISPLAY.* CLASS(OPERCMDS) ID(IZUSVR) ACCESS(READ)
 PERMIT MVS.DISPLAY.* CLASS(OPERCMDS) ID(SUPZOS) ACCESS(READ)

 RDEFINE SERVAUTH EZB.NETSTAT.**.VIPADCFG UACC(NONE)

 PERMIT EZB.NETSTAT.**.VIPADCFG +
   CLASS(SERVAUTH) ID(IZUSVR) ACCESS(READ)

 RDEFINE SERVAUTH EZB.NETWORKUTILS.CLOUD.**        UACC(NONE)

 PERMIT EZB.NETWORKUTILS.CLOUD.**        CLASS(SERVAUTH) +
   ID(IZUSVR) ACCESS(READ)

 RDEFINE OPERCMDS MVS.ROUTE.CMD.** UACC(NONE)
 PERMIT MVS.ROUTE.CMD.**       CLASS(OPERCMDS) +
   ID(IZUSVR) ACCESS(READ)

 SETROPTS RACLIST(OPERCMDS,SERVAUTH) REFRESH
